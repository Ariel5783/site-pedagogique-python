
<TeacherOnly title="Corrigé Exercice 5">

### 1. Vérification parité

```python
def est_pair(n: int) -> bool:
    """
    Vérifie si un nombre est pair en testant le bit de poids faible.
    Un nombre est pair si son LSB (bit de droite) vaut 0.
    """
    return (n & 1) == 0

# Tests
assert est_pair(4) == True   # 0100 : LSB = 0
assert est_pair(7) == False  # 0111 : LSB = 1
assert est_pair(0) == True   # 0000 : LSB = 0
```

### 2. Manipulation de flags

```python
class Options:
    LECTURE = 1      # 0001
    ECRITURE = 2     # 0010
    EXECUTION = 4    # 0100
    ADMIN = 8        # 1000
    
    def __init__(self):
        self.flags = 0
    
    def activer(self, option: int) -> None:
        """Active une option avec OR"""
        self.flags |= option
    
    def desactiver(self, option: int) -> None:
        """Désactive une option avec AND NOT"""
        self.flags &= ~option
    
    def est_active(self, option: int) -> bool:
        """Vérifie si une option est active avec AND"""
        return bool(self.flags & option)

# Exemple d'utilisation
opts = Options()
opts.activer(Options.LECTURE | Options.ECRITURE)
print(f"Flags: {bin(opts.flags)}")  # 0b11

print(opts.est_active(Options.LECTURE))    # True
print(opts.est_active(Options.EXECUTION))  # False

opts.desactiver(Options.ECRITURE)
print(f"Flags: {bin(opts.flags)}")  # 0b1

opts.activer(Options.ADMIN)
print(f"Flags: {bin(opts.flags)}")  # 0b1001
```
